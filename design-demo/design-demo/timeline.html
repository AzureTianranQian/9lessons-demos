Next, the javascript...
- timeline javascript object has these properties:
-current.month (int)
-current.year (int)
-dates (key-value-pair object; keys are years in format yyyy, value is an array of arrays representing months
-goToMonthYear (function)
A day with an event receives the 'has-event' css class, and if the day on the timeline is hovered over, a list of the events will appear as a tooltip. Just the name is shown in this demo, but you can see how easily you could expand upon this.
					
$(document).ready(function() {
	/* Timeline */
	var timeline = {
		'current':{
			'month':'',
			'year':Date.today().getFullYear()
		},
		'dates':{},
		'goToMonthYear':function(mon,yr) {
			var direction = (mon+yr) < (this.current.month + this.current.year) ? '100%' : '-100%';
			var altdirection = direction == '100%' ? '-=100%' : '+=100%';
			this.current.month = mon;
			this.current.year = yr;
			var this_months_days = this.dates[yr][mon];
			$(".timeline-line").animate({left:direction},250,function() {
				var $days = [];
				for (var day in this_months_days) {
					var display_day = parseInt(day)+1;
					var day_events = this_months_days[day];
					var day_classes = 'timeline-day';
					if (day_events.length) {
						day_classes += ' has-event';
					}
					var day_of_week = Date.today().set({month:mon,day:display_day,year:yr}).toString("ddd");
					var $day = '
'+ day_of_week.substr(0,1) + '' + display_day + '
'; 
					$days.push($day);
				}
				$(".timeline-line").html($days.join("\r\n")).animate({left:altdirection},250,function() {
					$(".timeline-nav-icon").removeClass('disabled');
				})
				$(".timeline-monthyear-txt").text(Date.parse(mon+1 + "/" + yr).toString("MMMM") + " " + yr);						
			});
		}
	};
	
	//Build out date arrays
	//This year & next year
	var this_year = Date.today().getFullYear();
	var next_year = this_year + 1;
	var years_to_process = [this_year,next_year];
	for (var x = 0, yr; yr = years_to_process[x]; x++) {
		timeline.dates[yr] = {};
		var this_month = 0;
		
		for (var i = this_month; i < 12; i++) {
			timeline.dates[yr][i] = [];
			var days_in_month = Date.getDaysInMonth(yr,i);
			for (var n = 0; n < days_in_month; n++) {
				timeline.dates[yr][i].push([]);
			}
		}
		
	}
	
	//place current months days
	var this_month = Date.today().getMonth();
	timeline.goToMonthYear(this_month,this_year);
	
	//set up event handlers
	$(".timeline-nav-back").click(function() {
		if (!$(this).hasClass('disabled')) {
			$(this).addClass('disabled');
			if (timeline.current.month == 0) {
				//check if exists
				if (timeline.dates[timeline.current.year-1] != undefined && timeline.dates[timeline.current.year-1][11] != undefined) {
					timeline.goToMonthYear(12,timeline.current.year - 1);
				}
			}
			else {
				timeline.goToMonthYear(timeline.current.month - 1, timeline.current.year);
			}
		}
	});
	$(".timeline-nav-fwd").click(function() {
		if (!$(this).hasClass('disabled')) {
			$(this).addClass('disabled');
			if (timeline.current.month == 11) {
				//check if exists
				if (timeline.dates[timeline.current.year+1] != undefined && timeline.dates[timeline.current.year+1][0] != undefined) {
					timeline.goToMonthYear(0,timeline.current.year + 1);
				}
			}
			else {
				timeline.goToMonthYear(timeline.current.month + 1, timeline.current.year);
			}
		}
	});
	
	$(document).delegate(".timeline-day",{
		mouseenter:function() {
			var day = parseInt($(this).find('.bottom').text()) - 1;
			var days_events = timeline.dates[timeline.current.year][timeline.current.month][day];
			if (days_events != undefined && days_events.length) {
				var events_html = ['
    '];
				for (var i = 0, evt; evt = days_events[i]; i++) {
					events_html.push('
  • '+evt['name']+'
    • ');
				}
				events_html.push('
');
				var day_top = $(this).position().top + $(this).height() + 40;
				var day_left = $(this).position().left + ($(this).width() / 2);
				$(".timeline-hover").html(events_html.join("\r\n")).css({top:day_top,left:day_left}).fadeIn(250);
			}
		},
		mouseleave:function() {
			$(".timeline-hover:visible").fadeOut(250);
		}
	});
});
Now, the important part, which is adding items, or events, to the calendar. The items are properties of an individual day in the timeline.dates object, i.e. to check the number of items on November 6, 2012, just check timeline.dates[2012][10][6].length
Here is an example of how to feed a JSON source into timeline.dates
					
$.getJSON("events.json",function(r) {
	//Fill in proposals to dates
	var events = r;
	for (var i = 0, evt; evt = events[i]; i++) {
		var split_date = evt.date.split('/');
		var _mon = parseInt(split_date[0]) - 1;
		var _day = parseInt(split_date[1]) - 1;
		var _yr = split_date[2];
		timeline.dates[_yr][_mon][_day].push(evt);
		$("#timeline-"+_mon+"-"+(_day+1)).addClass('has-event');
	}
});
